\paragraph{Notation and hypotheses.}
Assume \(\gamma_1>0\) and
\(r_1,\widehat G_1,p^\pm
\in\widetilde{\mathcal E}(\mathbb R_+,V(\mathbb R))\), with
\(\widehat G_1\) independent of the Mellin covariable. Put
\[
\begin{aligned}
 a_{+r,\gamma_1}(x,\lambda)
 &:=p^+(\lambda)r_1(x/\gamma_1,\lambda),\\
 a_{+rG,\gamma_1}(x,\lambda)
 &:=p^+(\lambda)r_1(x/\gamma_1,\lambda)
    \widehat G_1(x/\gamma_1).
\end{aligned}
\]

\paragraph{Unweighted reductions.}
All four relations below are modulo compact operators:
\[
\begin{aligned}
 P^-\operatorname{Op}(r_1)D_{\gamma_1}
 &\simeq
 \operatorname{Op}_{\mathrm{right}\text{-}\gamma_1}
 (p^-r_1,d_{\gamma_1}),\\
 D_{\gamma_1}^{-1}P^+\operatorname{Op}(r_1)D_{\gamma_1}
 &\simeq \operatorname{Op}(a_{+r,\gamma_1}),\\
 P^-\operatorname{Op}(r_1)M_{\widehat G_1}
 &\simeq \operatorname{Op}(p^-r_1\widehat G_1),\\
 D_{\gamma_1}^{-1}P^+\operatorname{Op}(r_1)M_{\widehat G_1}
 &\simeq
 \operatorname{Op}_{\mathrm{right}\text{-}\gamma_1^{-1}}
 (a_{+rG,\gamma_1},d_{\gamma_1^{-1}}).
\end{aligned}
\]

\paragraph{Weighted formulas.}
Conjugating by \(\Phi_\delta^{-1}(\cdot)\Phi_\delta\) gives
\[
\begin{aligned}
 P^-R_1U_1
 &\simeq\Phi_\delta^{-1}
 \operatorname{Op}_{\mathrm{right}\text{-}\gamma_1}
 (p^-r_1,d_{\gamma_1})\Phi_\delta,\\
 U_1^{-1}P^+R_1U_1
 &\simeq\Phi_\delta^{-1}
 \operatorname{Op}(a_{+r,\gamma_1})\Phi_\delta,\\
 P^-R_1\widehat G_1
 &\simeq\Phi_\delta^{-1}
 \operatorname{Op}(p^-r_1\widehat G_1)\Phi_\delta,\\
 U_1^{-1}P^+R_1\widehat G_1
 &\simeq\Phi_\delta^{-1}
 \operatorname{Op}_{\mathrm{right}\text{-}\gamma_1^{-1}}
 (a_{+rG,\gamma_1},d_{\gamma_1^{-1}})\Phi_\delta.
\end{aligned}
\]

\paragraph{Residuals and status.}
The residuals are, respectively,
\[
 K_{-,r}D_{\gamma_1},\quad
 D_{\gamma_1}^{-1}K_{+,r}D_{\gamma_1},\quad
 P^-K_{r,G}+K_{-,rG},\quad
 D_{\gamma_1}^{-1}(P^+K_{r,G}+K_{+,rG}).
\]
They are compact by the two-sided ideal property and finite-sum closure.
The second and third outputs have standard
\(\widetilde{\mathcal E}\) symbols. The first and fourth are only
right-factorized Mellin representations; no standard membership of their
oscillatory products is claimed.

\paragraph{Boundary of the result.}
The factor \(W_{1,2}^{+}\) is not included or identified. No conclusion about
the complete correction, algebra membership of the pivot, or Fredholmness is
asserted.
